🚀 Основы JavaScript - Синтаксис

1. Переменные

Переменные - это контейнеры для хранения данных. В JavaScript есть три способа объявления переменных:

// Объявление переменных
let name = "Иван";           // можно изменять
const age = 25;              // нельзя изменять
var oldWay = "старый способ"; // устаревший способ

console.log("Имя:", name);
console.log("Возраст:", age);
console.log("Старый способ:", oldWay);

💡 Объяснение:

  • let - переменная, которую можно изменять
  • const - константа, которую нельзя изменять
  • var - старый способ (лучше не использовать)

2. Типы данных

JavaScript имеет несколько основных типов данных:

// Строки (текст)
let firstName = "Анна";
let lastName = 'Петрова';
let fullName = `${firstName} ${lastName}`; // шаблонные строки

// Числа
let integer = 42;
let decimal = 3.14;
let negative = -10;

// Булевы значения (true/false)
let isStudent = true;
let isWorking = false;

// Undefined и null
let notDefined;
let emptyValue = null;

console.log("Строки:", firstName, lastName, fullName);
console.log("Числа:", integer, decimal, negative);
console.log("Булевы:", isStudent, isWorking);
console.log("Особые:", notDefined, emptyValue);

3. Операторы

Операторы позволяют выполнять различные операции с данными:

// Математические операторы
let a = 10;
let b = 3;

console.log("Сложение:", a + b);        // 13
console.log("Вычитание:", a - b);       // 7
console.log("Умножение:", a * b);       // 30
console.log("Деление:", a / b);         // 3.333...
console.log("Остаток от деления:", a % b); // 1
console.log("Возведение в степень:", a ** b); // 1000

// Операторы сравнения
console.log("a > b:", a > b);           // true
console.log("a < b:", a < b);           // false
console.log("a == b:", a == b);         // false
console.log("a === b:", a === b);       // false (строгое сравнение)
console.log("a != b:", a != b);         // true
console.log("a !== b:", a !== b);       // true

// Логические операторы
let x = true;
let y = false;

console.log("И (&&):", x && y);         // false
console.log("ИЛИ (||):", x || y);       // true
console.log("НЕ (!):", !x);             // false

4. Условные конструкции

Условные конструкции позволяют выполнять разный код в зависимости от условий:

let score = 85;

// Простое условие if-else
if (score >= 90) {
    console.log("Отлично! Оценка A");
} else if (score >= 80) {
    console.log("Хорошо! Оценка B");
} else if (score >= 70) {
    console.log("Удовлетворительно! Оценка C");
} else {
    console.log("Неудовлетворительно! Оценка F");
}

// Тернарный оператор
let result = score >= 60 ? "Сдал" : "Не сдал";
console.log("Результат:", result);

// Switch для множественных условий
let day = "понедельник";
switch (day) {
    case "понедельник":
        console.log("Начинается рабочая неделя");
        break;
    case "пятница":
        console.log("Ура! Пятница!");
        break;
    case "суббота":
    case "воскресенье":
        console.log("Выходные!");
        break;
    default:
        console.log("Обычный рабочий день");
}

5. Функции

Функции - это блоки кода, которые можно переиспользовать:

// Объявление функции
function greet(name) {
    return `Привет, ${name}!`;
}

// Вызов функции
console.log(greet("Мария"));

// Функция с несколькими параметрами
function add(a, b) {
    return a + b;
}

console.log("Сумма 5 + 3 =", add(5, 3));

// Стрелочная функция (современный синтаксис)
const multiply = (x, y) => x * y;
console.log("Произведение 4 * 6 =", multiply(4, 6));

// Функция без параметров
const getRandomNumber = () => Math.floor(Math.random() * 100);
console.log("Случайное число:", getRandomNumber());
